3 Process

pc <- prcomp(t(vst))
percent <- summary(pc)$importance[2,]

3.1 PC1

message(sprintf("Component 1 explains %s %% of the variance",round(percent[1]*100)))
## Component 1 explains 44 % of the variance

The loadings are uniformely distributed

plot(density(pc$rotation[,1]),main="PC1 loadings distribution")
abline(v=cutoffs,lwd=2,lty=2,col="grey")

Selecting genes at the extremity of the distribution (mean +/- 3 sd, assuming a standard distribution)

cutoffs <- mean(pc$rotation[,1]) + (3 * sd(pc$rotation[,1])) * c(-1,1)
sel <- pc$rotation[,1] < cutoffs[1] | pc$rotation[,1] > cutoffs[2]

message(sprintf("There are %s genes in the selection",sum(sel)))
## There are 703 genes in the selection
pc1 <- rownames(vst[sel,])
write(pc1,file = "PCA/PC_1_gene.txt")
